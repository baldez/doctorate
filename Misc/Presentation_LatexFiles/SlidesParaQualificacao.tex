\begin{frame}
\begin{block}{Problema de agrupamento por partição}
Seja $\X = \{ \x_1, \dots, \x_m \}$, onde $\x_i = (x_{i1},\cdots,x_{id})  \in \R^d$. O problema de agrupamento consiste em obter uma partição $\mathbf{P}^*$ de $\X$ em $k$ subconjuntos que atenda um determinado critério de qualidade $Q$, de forma que $Q(\mathbf{P}^*)\geq Q(\mathbf{P}), \,\forall \mathbf{P}\in \mathcal{P}$. 
\end{block}


\pause

\begin{block}{Problema de agrupamento baseado em centroides}
Determinar um conjunto $C^* = \{\vetor{c}_1, \cdots, \vetor{c}_k\}$ de centroides que formem $k$ subconjuntos do tipo 
$$
C_j = \left\{\forall x \in X \Big| 
\argmin{1 \leq l \leq k \qquad\quad}{d(\vetor{x},\vetor{c}_{l})} = j \right\},
$$ 
de forma que $C^*$ atenda um critério de qualidade $Q$ tal que $Q(\mathbf{C}^*)\geq Q(\mathbf{C})$ para todo $C$.
\end{block}
\end{frame}

%%%   SLIDE   %%%%%%%%
\begin{frame}{Formalização do problema}

{\Large
\begin{block}{}
\begin{align}
\min &\,\;f(C) = \sum_{i=1}^{n} \min_{1 \leq j \leq k} ||x_i -c_j||^2 \nonumber\\
\text{s.a}\;\;& \,\; C \in \Omega \nonumber
\end{align}
\end{block}
}

\bigskip
tal que: 
\begin{enumerate}
\item $C \in \Omega \Rightarrow \left(\forall \vetor{c_j} \in C, \exists \vetor{x} \in \X \text{ tal que } j = \argmin{1 \leq l \leq k\qquad\qquad}{||x - c_l||^2_2}\right)$.

\item $\left(C_i, C_j \in \Omega \text{, com } C_i \neq C_j \right) \Rightarrow \vetor{x} \in C_i$ e $\vetor{y} \in C_j$, então $\exists \vetor{w} \in \R^d \text{ tal que } (\vetor{w}^t\vetor{x}) \cdot (\vetor{w}^t \vetor{y}) < 0$ 
\end{enumerate}
\end{frame}

\begin{frame}{Função \func{LocalSearch}}

\begin{algorithm}[H]
\caption{\kmeans\, projetado}
\AlgInit

\BlankLine
$\vetor{C} \leftarrow$ LocalSearch($\X, k, \vetor{C}, \Omega_c$)\\

{\small
\begin{description}
\item{\bf passo 1 (inicialização): } usar $C$ como solução inicial

\item{\bf passo 2 (atribuição): } associar cada elemento $x_j$, com $j\in \{1, \cdots,n\}$, ao centroide mais próximo ($C \rightarrow P$)

\item{\bf passo 3 (teste de otimalidade local): } se não houve alguma mudança no passo anterior, parar aqui

\item{\textcolor{red}{\bf passo 4 (projeção): }} \textcolor{red}{$C' \longleftarrow$ \func{Projection}$(C', C, \Omega_c)$}\\

\item{\bf passo 5 (atualização): } trocar os centroides $C$ da solução corrente por $C'$ e voltar para o {\bf passo 2}
\end{description}
}
\end{algorithm}
\IncMargin{2em}
\end{frame}

\begin{frame}{Espaço factível}
$\Omega_c = \{\mathcal{H}_1, \cdots, \mathcal{H}_{k'}\}$, com:
\begin{align}
\mathcal{H}_i = \left\{\x \in \R^d \Bigg| \mu_j - \frac{3\sigma_j}{\sqrt{n_i}} \leq x_j \leq \mu_j - \frac{3\sigma_j}{\sqrt{n_i}}, \forall j \in \{1, \cdots, d \}\right\},
\nonumber
\end{align}
onde $\mu_j$ e $\sigma_j$ são, respectivamente, a média e o desvio padrão, na dimensão j, da amostra $A_i = \{\x_1^{(i)}, \cdots, \x_{n_i}^{(i)}\}$ de uma classe $i$. 
\end{frame}

\subsection{Estruturas de vizinhança}
%%%   SLIDE   %%%%%%%%
\begin{frame}{Estrutura de vizinhança $1$}{realocação dos mais distantes}
{\Large
\begin{block}{}
\begin{align}
N^{(1)}_i (C_1) = \left\{C \in \Omega \Big| \left( r(C_1,C) = i \right) \bigwedge \left(\mathbf{w}(C_1) \cdot \mathbf{e}_j \geq L\right) \right\},
\nonumber
\end{align}
\end{block}
}
\bigskip
com $j$ varia entre $1$ e $i$.

\end{frame}

%%%   SLIDE   %%%%%%%%
\begin{frame}{Medida de aderência}

{\Large
\begin{block}{}
\begin{align}
\eta(\x_j) &= \dfrac{||\x_j - \vetor{c}_l|| - \mu_l}{\sigma_l},
\nonumber
\end{align}
\end{block}
}

\bigskip
onde
\begin{align} 
\mu_l &= \dfrac{\sum_{\x \in \mathbf{C}_l} ||\x - \vetor{c}_l||}{\#\mathbf{C}_l}, %&(\text{média das distorções do grupo } l)\label{media}
\nonumber
\\
\sigma_l &= \left( \frac{1}{\#\mathbf{C}_l}\sum_{\x \in \mathbf{C}_l} \left| ||\x_j - \vetor{c}_l|| - \mu_l\right|^2 \right)^{\frac{1}{2}}. 
\nonumber
%&(\text{desvio padrão das distorções do grupo } l)\label{desviopadrao}
\end{align}
\end{frame}

%%%   SLIDE   %%%%%%%%
\begin{frame}{Estrutura de vizinhança $2$}{realocação do mais discrepante}
{\Large
\begin{block}{}
\begin{align}
N^{(2)}_i (C_1) = \left\{C \in \Omega \Big| (r(C_1,C) = i) \bigwedge \left( \Theta(C_1) \cdot \mathbf{e}_j \geq L, \right)\right\}
\nonumber
\end{align}
\end{block}
}

\bigskip
\noindent onde: 
\begin{itemize}
\item $j$ varia entre $1$ e $i$,
\item $L$ é o $i$-ésimo maior valor de $\eta$ entre os elementos da solução $C_1$,
\item $\Theta(C_1)$ é o vetor decrescente dos valores $\eta$ dos elementos de $C_1$.
\end{itemize}
\end{frame}

%%%   SLIDE   %%%%%%%%
\begin{frame}{Estrutura de vizinhança $3$}{realocação intensa}

{\Large
\begin{block}{}
\begin{align}
 N_i^{(3)} (C_1) = 
\left\{
C \in \Omega \Big|
\Theta(C_i) \cdot \mathbf{e}_j \geq L_i
\right\},
\nonumber
\end{align}
\end{block}
\bigskip
\hfill com $L_i = 5 - 0,1i$.
}
\end{frame}

\begin{frame}
Para $y(k)$, com $k \in \{1,2,3, \cdots , n \}$, têm-se:

\begin{align}
y_t &= \bar{y}+ \sum^{\lfloor\frac{n}{2}\rfloor}_{k=1}\left[ A_k\cos \left( \frac{2\pi kt}{n}\right) + B_k\sin\left(\frac{2\pi kt}{n}\right)\right],
\nonumber
\end{align}
onde $\bar{y}$ é a média aritmética dos dados e
\begin{eqnarray}
A_k = \frac{2}{n}\sum^n_{t=1}y_t\cos\left( \frac{2\pi kt}{n}\right) \qquad\textrm{e}\qquad
B_k = \frac{2}{n}\sum^n_{t=1}y_t\sin\left( \frac{2\pi kt}{n}\right).
\nonumber
\end{eqnarray}
\end{frame}
